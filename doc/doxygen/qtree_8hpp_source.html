205 template <class T> void Qtree<T>::add(const T &t, const Coord &coord) {
206  minx = MIN(minx, coord.x);
207  maxx = MAX(maxx, coord.x);
208  miny = MIN(miny, coord.y);
209  maxy = MAX(maxy, coord.y);
210  if(this->child->contains(coord))
211  this->child->add(t, coord);
212  else {
213  /* add would be out of bounds so create
214  a new child and double the size */
215  QtreeNode<T> *oc = this->child;
216  int ns;
217  ns = oc->size;
218  Coord newpos = oc->coord;
219  if(coord.x < newpos.x)
220  newpos.x -= ns;
221  if(coord.y < newpos.y)
222  newpos.y -= ns;
223  ns *= 2;
224  this->child = new QtreeNode<T>(newpos, ns*2);
225  this->child->add(oc);
226  /* tail call recursion */
227  this->add(t, coord);
228  }
229 }

345 #ifdef DEBUG
346  if(!this->contains(coord))
347  throw std::out_of_range("tried to add an out of range value to QtreeNode");
348 #endif
349  if(this->size == 1) {
350  bool ret = !this->value_set;
351  this->value = t;
352  this->value_set = true;
353  this->memberCount = 1;
354  return ret;
355  }
356  fill_content();
357  for(int i = 0; i < 4; i++)
358  if(this->content[i]->contains(coord)) {
359  bool ret = this->content[i]->add(t, coord);

361  this->memberCount++;
362  return ret;
363  }
